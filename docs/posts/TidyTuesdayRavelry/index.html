# ------ (creative) DATA VISUALIZATION ------

df_dataviz <- df %>%
  group_by(yarn_company_name, yarn_weight_name) %>%
  summarise(avg_rating_average = mean(rating_average),
            avg_rating_count = mean(rating_count)) %>%
  ungroup()
